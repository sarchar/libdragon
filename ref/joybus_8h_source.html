252inline void joybus_exec_cmd(
253 int port,
254 size_t send_len,
255 size_t recv_len,
256 const void *send_data,
257 void *recv_data
258)
259{
260 // Validate the desired Joybus port offset
261 assert((port >= 0) && (port < JOYBUS_PORT_COUNT));
262 // Ensure the send_len and recv_len fit in the Joybus operation block
263 assert((port + send_len + recv_len) < (JOYBUS_BLOCK_SIZE - 4));
264 // Allocate the Joybus operation block input and output buffers
265 uint8_t input[JOYBUS_BLOCK_SIZE] = {0};
266 uint8_t output[JOYBUS_BLOCK_SIZE] = {0};
267 // Skip commands on ports before the desired port offset
268 size_t i = port;
269 // Set the command metadata
270 input[i++] = send_len;
271 input[i++] = recv_len;
272 // Copy the send_data into the input buffer
273 memcpy(&input[i], send_data, send_len);
274 i += send_len + recv_len;
275 // Close out the Joybus operation block
276 input[i] = 0xFE;
277 input[sizeof(input) - 1] = 0x01;
278 // Execute the Joybus operation synchronously
279 joybus_exec(input, output);
280 // Copy recv_data from the output buffer
281 memcpy(recv_data, &output[i - recv_len], recv_len);
282}
